#!/usr/bin/env python3 """ 기상할증 알리미 — 데스크톱 위젯 실행: python3 weather_widget_gui.py 종료: 우클릭 """ import tkinter as tk from tkinter import font as tkfont import json, urllib.request, threading, time, subprocess, os from datetime import datetime # ── 색상 ─────────────────────────────────────────────────── BG = '#12161E' CARD = '#141921' SEP = '#1E2A38' GREEN = '#05CD82' ORANGE = '#F97316' BLUE = '#60A5FA' DIM = '#4A5568' TEXT = '#E2E8F0' # ── macOS 알림 ────────────────────────────────────────────── def mac_notify(title, message): try: script = f'display notification "{message}" with title "{title}" sound name "Ping"' subprocess.run(['osascript', '-e', script], check=False) except Exception: pass # ── 날씨 fetch (Open-Meteo) ──────────────────────────────── def fetch_weather(lat, lon): url = (f'https://api.open-meteo.com/v1/forecast' f'?latitude={lat}&longitude={lon}' f'¤t=temperature_2m,relative_humidity_2m,precipitation' f',weather_code,wind_speed_10m,apparent_temperature' f'&timezone=Asia%2FSeoul') with urllib.request.urlopen(url, timeout=10) as r: data = json.loads(r.read()) c = data['current'] code = c['weather_code'] temp = c['temperature_2m'] precip = c['precipitation'] humid = c['relative_humidity_2m'] isRain = code in RAIN_CODES or precip > 0 surge = isRain or temp <= -5 or temp >= 33 return { 'temp': temp, 'precip': precip, 'humid': humid, 'code': code, 'isRain': isRain, 'surge': surge, 'label': CODE_LABELS.get(code, '알 수 없음'), } # ── 메인 위젯 ────────────────────────────────────────────── class WeatherWidget: def __init__(self, region): self.region = region self.lat, self.lon = REGIONS[region] self.last_surge = None self.last_change = None self._build() self.root.after(500, self._fetch) self.root.after(600000, self._timer_fetch) self.root.mainloop() def _timer_fetch(self): self._fetch() self.root.after(600000, self._timer_fetch) def _build(self): self.root = tk.Tk() self.root.title('기상할증 알리미') self.root.attributes('-topmost', True) self.root.configure(bg=BG) self.root.geometry('+60+60') self.root.resizable(False, False) self.root.bind('', lambda e: self.root.destroy()) # 드래그 self.root.bind('', lambda e: setattr(self, '_dx', e.x) or setattr(self, '_dy', e.y)) self.root.bind('', lambda e: self.root.geometry(f'+{self.root.winfo_x()+e.x-self._dx}+{self.root.winfo_y()+e.y-self._dy}')) f = tk.Frame(self.root, bg=BG, padx=16, pady=14) f.pack() # 헤더 hdr = tk.Frame(f, bg=BG) hdr.pack(fill=tk.X, pady=(0,10)) tk.Label(hdr, text='● 기상할증 알리미', bg=BG, fg=GREEN, font=('Helvetica', 11, 'bold')).pack(side=tk.LEFT) tk.Label(hdr, text=self.region, bg=BG, fg=DIM, font=('Helvetica', 11)).pack(side=tk.RIGHT) tk.Frame(f, bg=SEP, height=1).pack(fill=tk.X, pady=(0, 10)) # 상태 카드 self.card = tk.Frame(f, bg='#0A2A1A', bd=0, padx=12, pady=12) self.card.pack(fill=tk.X, pady=(0,10)) self.lbl_icon = tk.Label(self.card, text='🌥', bg='#0A2A1A', font=('Helvetica', 36)) self.lbl_icon.pack() self.lbl_status = tk.Label(self.card, text='확인 중...', bg='#0A2A1A', fg=GREEN, font=('Helvetica', 15, 'bold')) self.lbl_status.pack(pady=(4,2)) self.lbl_sub = tk.Label(self.card, text='', bg='#0A2A1A', fg=GREEN, font=('Helvetica', 11)) self.lbl_sub.pack() self.lbl_change = tk.Label(self.card, text='', bg='#0A2A1A', fg=DIM, font=('Helvetica', 10)) self.lbl_change.pack(pady=(6,0)) tk.Frame(f, bg=SEP, height=1).pack(fill=tk.X, pady=(0,10)) # 수치 nums = tk.Frame(f, bg=BG) nums.pack(fill=tk.X, pady=(0,10)) self.lbl_temp = self._num(nums, '--°', '기온') self.lbl_precip = self._num(nums, '--', '강수량') self.lbl_humid = self._num(nums, '--%', '습도') # 푸터 self.lbl_footer = tk.Label(f, text='', bg=BG, fg='#2A3545', font=('Helvetica', 9)) self.lbl_footer.pack() tk.Label(f, text='우클릭 종료 • 드래그 이동', bg=BG, fg='#1E2A38', font=('Helvetica', 8)).pack(pady=(2,0)) def _num(self, parent, val, label): cell = tk.Frame(parent, bg=CARD, padx=14, pady=8) cell.pack(side=tk.LEFT, expand=True, fill=tk.X, padx=3) v = tk.Label(cell, text=val, bg=CARD, fg=TEXT, font=('Helvetica', 14, 'bold')) v.pack() tk.Label(cell, text=label, bg=CARD, fg=DIM, font=('Helvetica', 9)).pack() return v def _update(self, d): surge = d['surge'] isRain = d['isRain'] # 상태 변화 감지 → macOS 알림 if self.last_surge is not None and self.last_surge != surge: now_str = datetime.now().strftime('%H:%M') self.last_change = now_str + (' 비 시작' if surge else ' 비 그침') if surge: mac_notify(f'☔ {self.region} — 비 시작', '기상할증 ON을 권장해요.') else: mac_notify(f'☀️ {self.region} — 비 그침', '기상할증 OFF를 검토해주세요.') self.last_surge = surge # 색상 if surge: bg_c, fg_c = '#2A1A0A', ORANGE icon = '⚠️' st = '기상할증 ON 권장' sub = f'비가 내리고 있어요 · {d["precip"]}mm/h' if isRain else f'온도 기준 ({d["temp"]}°C)' else: bg_c, fg_c = '#0A2A1A', GREEN icon = '✅' st = '기상할증 OFF' sub = f'{d["label"]} · 할증 불필요' self.card.configure(bg=bg_c) for w in self.card.winfo_children(): w.configure(bg=bg_c) self.lbl_icon.configure(text=icon) self.lbl_status.configure(text=st, fg=fg_c) self.lbl_sub.configure(text=sub, fg=fg_c) if self.last_change: self.lbl_change.configure(text=self.last_change) self.lbl_temp.configure(text=f'{round(d["temp"],1)}°') self.lbl_precip.configure(text=f'{round(d["precip"],1)}mm') self.lbl_humid.configure(text=f'{round(d["humid"])}%') now_str = datetime.now().strftime('%H:%M') self.lbl_footer.configure(text=f'Open-Meteo · {now_str} 업데이트 · 10분 자동 갱신') def _fetch(self): def run(): try: d = fetch_weather(self.lat, self.lon) self.root.after(0, lambda: self._update(d)) except Exception as e: self.root.after(0, lambda: self.lbl_status.configure(text=f'오류: {str(e)[:30]}')) threading.Thread(target=run, daemon=True).start() def _start_timer(self): def loop(): while True: time.sleep(600) self._fetch() threading.Thread(target=loop, daemon=True).start() # ── 실행 ─────────────────────────────────────────────────── def start_widget(region): WeatherWidget(region) if os.path.exists(CONFIG_FILE): with open(CONFIG_FILE) as f: saved = f.read().strip() if saved in REGIONS: start_widget(saved) else: RegionSelector(start_widget) else: RegionSelector(start_widget)
